function sort012(arr){
  let left=0,right=arr.length-1,mid=0;
  while(left < right){
    if(arr[mid]===0){
      [arr[left],arr[mid]]=[arr[mid],arr[left]];
      left++;
      mid++;
    }
    else if (arr[mid]===1){
      mid++;
    }
    else{
      [arr[mid],arr[right]]=[arr[right],arr[mid]];
      right--;
    }
  }
  return arr;
}

let arr=[2,1,0,2,1,2,0];
let res=sort012(arr);
